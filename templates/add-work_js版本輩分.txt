<script>
  // 動態生成選項
  function createOptions(selectElement) {
    var options = [
      { value: '前期規劃', text: '前期規劃' },
      { value: '現場工程', text: '現場工程' },
      { value: '設備吊裝', text: '設備吊裝' },
      { value: '貨櫃預置工程', text: '貨櫃預置工程' },
      { value: '機電現場工程', text: '機電現場工程' },
      { value: '貨櫃現場工程', text: '貨櫃現場工程' },
      { value: '廠區現場工程', text: '廠區現場工程' }
    ];

    options.forEach(function(option) {
      var optionElement = document.createElement('option');
      optionElement.value = option.value;
      optionElement.text = option.text;
      selectElement.appendChild(optionElement);
    });
  }

  // 重新渲染LayUI的表格元件以顯示新的行
  layui.use('form', function() {
    var form = layui.form;
    var table = layui.table;
    table.init('option_value_table', {
      limit: 10
    });
    layui.form.render();
  });

  var newRowId = 0; // 新增行的ID計數器

  // 添加新的行
  document.getElementById('add_row_button').addEventListener('click', function() {
    var tableBody = document.getElementById('option_table_body');

    // 創建新的行
    var newRow = document.createElement('tr');
    newRow.setAttribute('data-new', 'true'); // 添加新行的標識屬性
    newRow.innerHTML = `
      <td></td>
      <td>
        <select id="select_${newRowId}" lay-filter="selectFilter" name="item"></select>
      </td>
      <td><input type="date" class="layui-input"></td>
      <td><input type="date" class="layui-input"></td>
      <td></td>
      <td><a href="javascript:void(0)" onclick="deleteRow(this)">刪除</a></td>
    `;

    // 為新行中的<select>元素動態生成選項
    var selectElement = newRow.querySelector(`#select_${newRowId}`);
    createOptions(selectElement);

    // 將新的行添加到表格中
    tableBody.appendChild(newRow);

    // 重新渲染LayUI的表格元件以顯示新的行
    layui.use('form', function() {
      var form = layui.form;
      var table = layui.table;
      table.init('option_value_table', {
        limit: 10
      });
      layui.form.render();
    });

    newRowId++; // 自增新增行的ID計數器
  });

  // 刪除行
  function deleteRow(element) {
    var row = element.parentNode.parentNode;
    row.parentNode.removeChild(row);
  }
</script>
